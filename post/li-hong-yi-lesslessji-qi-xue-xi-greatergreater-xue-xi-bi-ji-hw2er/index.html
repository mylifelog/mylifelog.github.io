# -*- coding: UTF-8 -*-
import numpy as np

# Load the dataset
train_x = np.loadtxt('../data/hw2/X_train', dtype='float', delimiter=',', skiprows=1)
train_y = np.loadtxt('../data/hw2/Y_train', dtype='float', delimiter=',', skiprows=1)
test_x = np.loadtxt('../data/hw2/X_test', dtype='float', delimiter=',', skiprows=1)

# 初始化参数矩阵
theta = np.ones(train_x.shape[1] + 1)
iteration = 100000
batch_num = 1600

#学习速率相关
lr = 0.01
use_adagrad = 1
lr_adagrad = 15

#一共32561行数据，预留561行作为test
test_size = 561+3200

# 定义sigmoid函数
# z: 输入的数组
# 输出每个元素的sigmoid求值结果
def sigmoid(z):
    e_tmp = np.zeros(len(z))
    e_tmp += np.e
    denomitor = 1 + np.power(e_tmp, np.asarray(z) * (-1))
    return 1 / denomitor

# 定义函数集
def F(theta, X_b):
    try:
        z = X_b.dot(theta.T)
        f_out = sigmoid(z)
        return f_out
    except:
        return float('inf')


# 定义一个通用的01分布 cross entropy函数
# 输入： 两个维度相同的一维数组
# 输出： 每一个对应位置求01分布的cross_entropy,返回相同size的数组
def cross_entropy(h, q):
    h = np.asarray(h)
    q = np.asarray(q)

    h_one = h + 1e-10
    h_zero = 1 - h + 1e-10

    return (q * (np.log(h_one)) + (1-q) * (np.log(h_zero))) * -1



# 定义损失函数CrossEntropy： 使用CrossEntropy计算一个batch数据的Loss
def Loss(f, y):
    try:
        f = np.asarray(f)
        y = np.asarray(y)
        cr = cross_entropy(f, y)
        return np.sum(cr) / len(y)
    except:
        return float('inf')

# 计算梯度矩阵: 1个batch的计算结果
def gradLoss(X_b, f, y):
    return X_b.T.dot(f - y) / len(y)
    # return X_b.T.dot(X_b.dot(theta) - y) * 2. / len(y)

# 计算训练数据的准确率
def calcTrainAcc():
    batch_num = len(train_x) - test_size
    n = 0
    X_b_test = np.hstack([train_x[n:n+batch_num], np.ones((len(train_x[n:n+batch_num]), 1))])
    #对X按列进行归一化
    #X_b_test = X_b_test / (X_b_test.max(axis=0) + 1e-10)
    y_test = train_y[n:n+batch_num]
    f_test = F(theta, X_b_test)
    f_test = np.around(f_test).astype('int32')
    r_test = np.abs(y_test - f_test)
    correct = 0
    for r in r_test:
        if r < 0.5:
            correct += 1
    return 100 * correct / batch_num
# 使用预留的test数据计算准确率
def calcTestAcc():
    batch_num = test_size
    n = len(train_x) - batch_num
    X_b_test = np.hstack([train_x[n:n+batch_num], np.ones((len(train_x[n:n+batch_num]), 1))])
    #对X按列进行归一化
    #X_b_test = X_b_test / (X_b_test.max(axis=0) + 1e-10)
    y_test = train_y[n:n+batch_num]
    f_test = F(theta, X_b_test)
    f_test = np.around(f_test).astype('int32')
    r_test = np.abs(y_test - f_test)
    correct = 0
    for r in r_test:
        if r < 0.5:
            correct += 1
    return 100 * correct / test_size

# 使用预留的test数据计算准确率
def calcTestResult():
    batch_num = len(test_x)
    n = 0
    X_b_test = np.hstack([test_x[n:n+batch_num], np.ones((len(test_x[n:n+batch_num]), 1))])
    #对X按列进行归一化
    X_b_test = X_b_test / (X_b_test.max(axis=0) + 1e-10)

    f_test = F(theta, X_b_test)
    #r_test = np.abs(y_test - f_test)
    f_test = np.around(f_test).astype('int32')
    f_index = np.zeros(f_test.shape[0])
    for i in range (len(f_index)):
        f_index[i] = i+1
    r_test = np.vstack([f_index.T,f_test.T]).astype('int32')
    np.savetxt("hw2_test_result.csv", r_test.T, fmt='%d', delimiter=',', header="id,label", comments="")

    #print(r_test)

#统一对数据做归一化
train_x = train_x / (train_x.max(axis=0) + 1e-10)

#adagrad临时变量
sum_g_sq = np.zeros(theta.shape[0])
for i in range(iteration):
    for n in range(0, len(train_x) - test_size, batch_num):
        #取一个batch的数据，并对数据集添加Bias
        X_b_bat = np.hstack([train_x[n:n+batch_num], np.ones((len(train_x[n:n+batch_num]), 1))])
        y_bat = train_y[n:n+batch_num]

        #对X按列进行归一化
        #X_b_bat = X_b_bat / (X_b_bat.max(axis=0) + 1e-10)

        #print(X_b_bat)

        #计算当前function结果
        f_bat = F(theta, X_b_bat)
        #使用function结果和y数据计算loss
        L = Loss(f_bat, y_bat)
        #计算梯度矩阵
        grad = gradLoss(X_b_bat, f_bat, y_bat)
        #更新参数矩阵
        if use_adagrad:
            #使用AdaGrad计算lengingRate: ln除以历史梯度平方和开根号
            sum_g_sq += grad**2 + 1e-10
            ada_lr = lr_adagrad * 1./np.sqrt(sum_g_sq)
            theta = theta - ada_lr*grad
        else:
            theta = theta - lr*grad
    if i % 10 == 0:
        print("T=",i)
        #calcTestResult()
        #print("Loss:", L)
        print("TrainAcc:", calcTrainAcc())
        print("TestAcc:", calcTestAcc())
        if (i == 7700):
            print("result saved!")
            calcTestResult()
